<script>

  // ── State

  let selectedCat = null;


  // ── Today

  const today = new Date();

  const dayNum = today.getDate();

  const dateLabel = today.toLocaleDateString('de-DE', { day: 'numeric', month: 'long', year: 'numeric' });

  document.getElementById('today-display').textContent = dateLabel;


  // ── Category buttons

  document.querySelectorAll('.cat-btn').forEach(btn => {

    btn.addEventListener('click', () => {

      document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));

      btn.classList.add('active');

      selectedCat = btn.dataset.cat;

      document.getElementById('cat-err').hidden = true;

    });

  });


  // ── Submit

  document.getElementById('submit-btn').addEventListener('click', () => {

    const betragRaw = document.getElementById('betrag').value.trim();

    const beschreibung = document.getElementById('beschreibung').value.trim();


    let valid = true;


    if (!betragRaw || isNaN(parseFloat(betragRaw)) || parseFloat(betragRaw) <= 0) {

      document.getElementById('betrag-err').hidden = false;

      valid = false;

    } else {

      document.getElementById('betrag-err').hidden = true;

    }


    if (!selectedCat) {

      document.getElementById('cat-err').hidden = false;

      valid = false;

    } else {

      document.getElementById('cat-err').hidden = true;

    }


    if (!valid) return;


    const betrag    = parseFloat(betragRaw).toFixed(2);

    const desc      = beschreibung || '—';

    const kategorie = selectedCat;

    const datum     = String(dayNum);


    const base = 'https://docs.google.com/forms/d/e/1FAIpQLSdJBCSdlctmVTxIrMywL6HysqsfINqAgEtve3icHsYp2tpUUg/formResponse';

    const params = new URLSearchParams({

      'entry.26907830':   betrag,

      'entry.514198582':  desc,

      'entry.1689769809': kategorie,

      'entry.1969182734': datum,

      'submit':           'Submit'

    });


    const url = `${base}?${params.toString()}`;


    // Submit via hidden iframe to avoid CORS/redirect issues

    let iframe = document.getElementById('submit-frame');

    if (!iframe) {

      iframe = document.createElement('iframe');

      iframe.id = 'submit-frame';

      iframe.name = 'submit-frame';

      iframe.style.display = 'none';

      document.body.appendChild(iframe);

    }


    const form = document.createElement('form');

    form.method = 'POST';

    form.action = base;

    form.target = 'submit-frame';


    const fields = {

      'entry.26907830':   betrag,

      'entry.514198582':  desc,

      'entry.1689769809': kategorie,

      'entry.1969182734': datum,

    };


    for (const [name, value] of Object.entries(fields)) {

      const input = document.createElement('input');

      input.type  = 'hidden';

      input.name  = name;

      input.value = value;

      form.appendChild(input);

    }


    document.body.appendChild(form);

    form.submit();

    document.body.removeChild(form);


    // Reset

    setTimeout(() => {

      document.getElementById('betrag').value = '';

      document.getElementById('beschreibung').value = '';

      document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));

      selectedCat = null;

      showToast();

    }, 300);

  });


  // ── Toast

  function showToast() {

    const t = document.getElementById('toast');

    t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 2800);

  }

</script>
